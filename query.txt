CREATE TABLE Questions (
    question_id INTEGER PRIMARY KEY,
    year INTEGER NOT NULL,
    subject NVARCHAR(255) NOT NULL,
    topic NVARCHAR(255),
    question_text TEXT NOT NULL,
    difficulty_level VARCHAR(50) DEFAULT 'medium'
);

CREATE TABLE Options (
    option_id INTEGER PRIMARY KEY AUTOINCREMENT,
    question_id INTEGER NOT NULL,
    option_text TEXT NOT NULL,
    is_correct BOOLEAN NOT NULL CHECK (is_correct IN (0, 1)),
    FOREIGN KEY (question_id) REFERENCES Questions(question_id)
);

CREATE TABLE Tags (
    tag_id INTEGER PRIMARY KEY AUTOINCREMENT,
    name NVARCHAR(255) UNIQUE NOT NULL
);

CREATE TABLE QuestionTags (
    question_id INTEGER NOT NULL,
    tag_id INTEGER NOT NULL,
    PRIMARY KEY (question_id, tag_id),
    FOREIGN KEY (question_id) REFERENCES Questions(question_id),
    FOREIGN KEY (tag_id) REFERENCES Tags(tag_id)
);

CREATE TABLE UserProgress (
    user_id INTEGER NOT NULL,
    question_id INTEGER NOT NULL,
    status VARCHAR(50) CHECK (status IN ('correct', 'wrong', 'unanswered')),
    attempts INTEGER DEFAULT 0,
    last_attempt_date DATETIME,
    score INTEGER,
    PRIMARY KEY (user_id, question_id),
    FOREIGN KEY (question_id) REFERENCES Questions(question_id)
);

CREATE TABLE Exams (
    exam_id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    mode VARCHAR(50) CHECK (mode IN ('simulation', 'topic_based', 'adaptive'))
);

CREATE TABLE ExamQuestions (
    exam_id INTEGER NOT NULL,
    question_id INTEGER NOT NULL,
    user_answer INTEGER, -- Corresponds to option_id
    is_correct BOOLEAN,
    PRIMARY KEY (exam_id, question_id),
    FOREIGN KEY (exam_id) REFERENCES Exams(exam_id),
    FOREIGN KEY (question_id) REFERENCES Questions(question_id)
);
